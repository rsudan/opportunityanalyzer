## FEATURE: ONE-PAGE PROJECT REPORT GENERATION

Add a "Generate Detailed Report" button that creates an elegantly formatted one-page report for each selected scored project, ordered by score (highest to lowest). The report should be downloadable as a styled HTML file that can be printed or saved as PDF.

// POST /api/detailed-report - Generate one-page report for selected projects
app.post('/api/detailed-report', async (req, res) => {
  const { projects } = req.body;

  // Sort projects by score (highest first)
  const sortedProjects = [...projects]
    .filter(p => p.score) // Only include scored projects
    .sort((a, b) => (b.score.overall_score || 0) - (a.score.overall_score || 0));

  if (sortedProjects.length === 0) {
    return res.status(400).json({ error: 'No scored projects to report on' });
  }

  // Generate HTML report
  const html = generateDetailedReport(sortedProjects);

  res.json({ html });
});

function generateProjectPage(project, rank) {
  const score = project.score;
  const overall = score.overall_score;

  const classification = overall >= 7 ? 'High Priority' : overall >= 5 ? 'Strong Potential' : 'Moderate Fit';

  // Format country
  const country = Array.isArray(project.countryname)
    ? project.countryname.join(', ')
    : project.countryname || 'Unknown';

  // Format amount
  const amount = formatAmount(project.totalamt);

  // Get dimension scores
  const techScore = score.emerging_tech?.score || 0;
  const foresightScore = score.foresight?.score || 0;
  const collectiveScore = score.collective_intelligence?.score || 0;

  // Get findings
  const technologies = score.emerging_tech?.technologies?.join(', ') || 'Not identified';
  const applications = score.emerging_tech?.applications?.join('; ') || 'N/A';
  const techEvidence = score.emerging_tech?.evidence || '';

  const disruptions = score.foresight?.disruptions?.join(', ') || 'Not identified';
  const horizon = score.foresight?.horizon || 'unknown';
  const foresightEvidence = score.foresight?.evidence || '';

  const ecosystemActivity = score.collective_intelligence?.ecosystem_activity || 'unknown';
  const examples = score.collective_intelligence?.examples?.join(', ') || 'Not identified';
  const collectiveEvidence = score.collective_intelligence?.evidence || '';

  const opportunities = score.top_opportunities || [];

  const keyInsight = score.key_insight || 'Comprehensive analysis required to identify specific opportunities.';

function extractDomain(projectName) {
  return projectName
    .replace(/project$/i, '')
    .replace(/additional financing/i, '')
    .replace(/phase [ivx0-9]+/i, '')
    .replace(/\([^)]*\)/g, '')
    .replace(/\s+/g, ' ')
    .trim();
}

const [detailedReportHtml, setDetailedReportHtml] = useState('');

async function generateDetailedReport() {
  const selectedProjects = projects.filter(p => selected.has(p.id));
  const scoredProjects = selectedProjects.filter(p => scores[p.id]);

  if (scoredProjects.length === 0) {
    alert('Please select and score at least one project first');
    return;
  }

  setGenerating(true);
  try {
    const projectsWithScores = scoredProjects.map(p => ({
      ...p,
      score: scores[p.id]
    }));

    const response = await fetch('/api/detailed-report', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ projects: projectsWithScores })
    });

    const data = await response.json();

    // Open report in new window
    const reportWindow = window.open('', '_blank');
    reportWindow.document.write(data.html);
    reportWindow.document.close();

  } catch (error) {
    alert('Report generation failed: ' + error.message);
  }
  setGenerating(false);
}
